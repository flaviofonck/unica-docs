@Async
public void procesarAsync(MultipartFile archivo, String jobId) {
    try (Workbook wb = WorkbookFactory.create(archivo.getInputStream())) {
        Sheet sheet = wb.getSheet("BASE GENERAL");
        if (sheet == null) throw new IllegalArgumentException("Hoja BASE GENERAL no encontrada");
        Row header = sheet.getRow(0);
        // Mapear índice de columna por nombre de encabezado
        Map<String, Integer> colIdx = new HashMap<>();
        for (Cell c : header) colIdx.put(c.getStringCellValue().trim(), c.getColumnIndex());
        int total = sheet.getLastRowNum();
        for (int i = 1; i <= total; i++) {
            Row row = sheet.getRow(i);
            // Leer solo las 14 columnas mapeadas
            // Campo clave: ID ENLACE (col 9). Cols 10 y 11 nombre duplicado — parsear por índice
            String idEnlace = getCellValue(row, colIdx.get("ID ENLACE"));
            String catResolucion = getCellValue(row, 9);  // por índice
            String motivoFallo   = getCellValue(row, 10); // por índice
            jobStatusService.actualizar(jobId, i, total);
        }
        jobStatusService.completar(jobId);
    } catch (Exception e) {
        jobStatusService.error(jobId, e.getMessage());
    }
}

@Async
public void importarFiscAsync(MultipartFile archivo, String jobId) {
    String nombre = archivo.getOriginalFilename();
    jobStatusService.init(jobId);
    try {
        List<String> lineas;
        if (nombre != null && nombre.toLowerCase().endsWith(".rar")) {
            lineas = extraerDesdeRar(archivo.getInputStream());   // junrar
        } else {
            lineas = extraerDesdeZip(archivo.getInputStream());   // commons-compress
        }
        // Extraer mes/año del nombre del archivo
        Pattern p = Pattern.compile(
            "(ENE|FEB|MAR|ABR|MAY|JUN|JUL|AGO|SEP|OCT|NOV|DIC)_(\\d{2})");
        Matcher m = p.matcher(nombre);
        if (!m.find()) throw new IllegalArgumentException("Nombre de archivo no contiene mes/año");
        String mes = MESES.get(m.group(1));   // "ENE" → "01"
        String anio = "20" + m.group(2);
        // Verificar control de idempotencia
        if (controlCargaFiscRepo.existsByMesAnio(mes, anio))
            throw new IllegalStateException("Ya existe carga para " + mes + "/" + anio);
        int total = lineas.size();
        for (int i = 0; i < total; i++) {
            procesarLineaFisc(lineas.get(i), mes, anio);
            jobStatusService.actualizar(jobId, i + 1, total);
        }
        jobStatusService.completar(jobId);
    } catch (Exception e) {
        jobStatusService.error(jobId, e.getMessage());
    }
}

@PostMapping("/migraciones/importar")
public ResponseEntity<?> importarMigraciones(
        @RequestParam("archivo") MultipartFile archivo) throws IOException {
    try (Workbook wb = WorkbookFactory.create(archivo.getInputStream())) {
        Sheet sheet = wb.getSheet("DATOS TECNICOS Y CONTACTOS");
        if (sheet == null) throw new IllegalArgumentException(
            "Hoja 'DATOS TECNICOS Y CONTACTOS' no encontrada");
        Row header = sheet.getRow(0);
        Map<String, Integer> colIdx = new HashMap<>();
        for (Cell c : header) colIdx.put(c.getStringCellValue().trim(), c.getColumnIndex());
        List<Migracion> registros = new ArrayList<>();
        for (int i = 1; i <= sheet.getLastRowNum(); i++) {
            Row row = sheet.getRow(i);
            if (row == null) continue;
            registros.add(mapearFila(row, colIdx));
        }
        migracionService.guardarYDetectarInconsistencias(registros);
        return ResponseEntity.ok(Map.of("importados", registros.size()));
    }
}

@PostMapping("/bajas/importar-salesforce")
public ResponseEntity<?> importarBajasSalesforce(
        @RequestParam("archivo") MultipartFile archivo) throws IOException {
    CSVParser parser = CSVFormat.DEFAULT
        .withDelimiter(';')
        .withFirstRecordAsHeader()
        .parse(new InputStreamReader(
            archivo.getInputStream(),
            StandardCharsets.ISO_8859_1));   // latin1 OBLIGATORIO
    List<BajaSalesforce> registros = new ArrayList<>();
    for (CSVRecord rec : parser) {
        // FUN: Nombre (col 2) vacío en datos reales — usar col 4
        String dsNombre = rec.get("Nombre de Descripción de servicio"); // DS-XXXXXXXX
        registros.add(mapear(rec));
    }
    bajasService.procesarBajasSalesforce(registros);
    return ResponseEntity.ok(Map.of("importados", registros.size()));
}

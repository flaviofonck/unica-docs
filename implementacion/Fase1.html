   private List<String> obtenerSeccionesVisibles(Usuario usuario) {
       return usuarioPerfilRepository.findByUsuario(usuario).stream()
           .flatMap(up -> permisoRepository.findByPerfil(up.getPerfil()).stream())
           .filter(p -> Boolean.TRUE.equals(p.getPuedeVer()))
           .map(Permiso::getSeccion)
           .filter(s -> s != null && !s.isBlank())
           .distinct()
           .toList();

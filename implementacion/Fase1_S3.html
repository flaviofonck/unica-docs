    G2 — indicador_iva = 0 para pais != COL (RN-10)¶

    HU-05 / Regla RN-10: Si la ciudad del servicio está fuera de Colombia (pais != 'COL') → indicador_iva = 0 (servicios internacionales no aplican IVA colombiano)

    Implementación requerida¶

    Servicio: ConfiguracionServicioService.java — método aplicarReglas(ConfiguracionServicio cs):

    /**
 * RN-10: Si la ciudad del servicio está fuera de Colombia → indicador_iva = 0
 */
private void aplicarReglaIndicadorIva(ConfiguracionServicio cs) {
    if (cs.getCodigoDaneCiudad() != null) {
        DimCiudad ciudad = dimCiudadRepository
            .findByCodigoDane(cs.getCodigoDaneCiudad())
            .orElse(null);
        if (ciudad != null && !"COL".equals(ciudad.getPais())) {
            cs.setIndicadorIva(BigDecimal.ZERO);
        }
    }
}

    Cuándo se invoca: al crear o modificar un dim_configuracion_servicios — antes de persistir.

    Test unitario requerido¶

    @Test
void aplicarReglas_ciudadInternacional_debeSetearIndicadorIvaCero() {
    // Arrange
    DimCiudad ciudadUSA = new DimCiudad();
    ciudadUSA.setCodigoDane("US-NYC");
    ciudadUSA.setPais("USA");

    ConfiguracionServicio cs = new ConfiguracionServicio();
    cs.setCodigoDaneCiudad("US-NYC");
    cs.setIndicadorIva(new BigDecimal("0.19"));

    when(dimCiudadRepository.findByCodigoDane("US-NYC"))
        .thenReturn(Optional.of(ciudadUSA));

    // Act
    service.aplicarReglas(cs);

    // Assert
    assertThat(cs.getIndicadorIva()).isEqualByComparingTo(BigDecimal.ZERO);
}

@Test
void aplicarReglas_ciudadColombia_noModificaIndicadorIva() {
    DimCiudad ciudadCOL = new DimCiudad();
    ciudadCOL.setCodigoDane("11001");
    ciudadCOL.setPais("COL");

    ConfiguracionServicio cs = new ConfiguracionServicio();
    cs.setCodigoDaneCiudad("11001");
    cs.setIndicadorIva(new BigDecimal("0.19"));

    when(dimCiudadRepository.findByCodigoDane("11001"))
        .thenReturn(Optional.of(ciudadCOL));

    service.aplicarReglas(cs);

    assertThat(cs.getIndicadorIva()).isEqualByComparingTo(new BigDecimal("0.19"));
}
